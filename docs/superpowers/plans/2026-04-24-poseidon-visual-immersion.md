> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement the v0.6.0 visual + immersion overhaul: per-theme ambient motion behind every menu, Triton omnipresence (4-bit indexed sprite + corner mascot + mood-tints-theme + narrative toasts), 8 procedural + 2 GIF screensavers with theme auto-pick.

**Architecture:** Purely additive. Three new modules (`ui_ambient`, `screensaver`, `screensavers/sav_*`), one macro redefine (`T_ACCENT` → effective-accent function), one main-loop call (`screensaver_tick`), one menu integration (`ui_ambient_tick` in `menu_draw_body`). Python 3 + Pillow for asset tooling. No unit test framework — verification is build → flash → visual check on hardware.

```cpp
/* ---- POSEIDON: deep-sea motes + slow wave-band ripple ---- */
static void amb_poseidon(int x, int y, int w, int h)
{
    auto &d = M5Cardputer.Display;
    uint32_t now = millis();
    float scale = ambient_speed_scale();

    /* 8 motes, each on its own slow vertical loop. Seed per-mote so they
     * don't sync. Each mote: 2px dot of theme.accent or theme.accent2,
     * rises bottom→top, fades in/out at endpoints. */
    static const struct { uint8_t col_pct; uint16_t period_ms; uint32_t seed; bool alt; } MOTES[8] = {
        {  8, 5000, 0x1111, false},
        { 22, 7000, 0x2222, true },
        { 36, 6000, 0x3333, false},
        { 50, 8000, 0x4444, true },
        { 62, 5500, 0x5555, false},
        { 78, 7000, 0x6666, false},
        { 88, 6500, 0x7777, true },
        { 14, 9000, 0x8888, false},
    };
    for (int i = 0; i < 8; ++i) {
        uint32_t period = (uint32_t)(MOTES[i].period_ms / scale);
        if (period < 500) period = 500;
        uint32_t phase = (now + MOTES[i].seed) % period;
        /* y goes from h (bottom) to -2 (off top) */
        int my = h - 2 - (int)((int64_t)phase * (h + 2) / period);
        int mx = (MOTES[i].col_pct * w) / 100;
        if (my < 0 || my >= h) continue;
        uint16_t color = MOTES[i].alt ? T_ACCENT2 : T_ACCENT;
        d.drawPixel(x + mx, y + my, color);
        /* tiny 1px halo above/below */
        if (my > 0)     d.drawPixel(x + mx, y + my - 1, color);
        if (my < h - 1) d.drawPixel(x + mx, y + my + 1, color);
    }

    /* Wave-band: one horizontal line whose y oscillates slowly. */
    uint32_t wave_period = (uint32_t)(8000.0f / scale);
    if (wave_period < 1000) wave_period = 1000;
    uint32_t wphase = now % wave_period;
    /* Triangle wave between h*0.1 and h*0.8 */
    int range = (h * 7) / 10;
    int wy = (h / 10) + (int)abs((int)(wphase * 2 * range / wave_period) - range);
    if (wy >= 0 && wy < h) {
        d.drawFastHLine(x, y + wy, w, T_ACCENT);
    }
}
```

```cpp
/* ---- AMBER: CRT scanline drift + phosphor jitter ---- */
static void amb_amber(int x, int y, int w, int h)
{
    auto &d = M5Cardputer.Display;
    uint32_t now = millis();
    float scale = ambient_speed_scale();

    /* Scanline drifting top→bottom. Period ~3s neutral. */
    uint32_t period = (uint32_t)(3000.0f / scale);
    if (period < 600) period = 600;
    uint32_t phase = now % period;
    int sy = (int)((int64_t)phase * (h + 8) / period) - 4;
    if (sy >= 0 && sy < h) {
        d.drawFastHLine(x, y + sy,     w, T_ACCENT);
    }
    if (sy + 1 >= 0 && sy + 1 < h) {
        /* dimmer trailing line */
        d.drawFastHLine(x, y + sy + 1, w, T_DIM);
    }

    /* Phosphor jitter — 4 random pixels per frame at low brightness */
    for (int i = 0; i < 4; ++i) {
        int px = x + (int)(esp_random() % w);
        int py = y + (int)(esp_random() % h);
        d.drawPixel(px, py, T_DIM);
    }
}
```

```cpp
/* ---- TRON: scrolling grid + packet hop ---- */
static void amb_tron(int x, int y, int w, int h)
{
    auto &d = M5Cardputer.Display;
    uint32_t now = millis();
    float scale = ambient_speed_scale();

    /* Grid: 30px cells, scrolling diagonally. Lines at theme.dim brightness. */
    uint32_t scroll_period = (uint32_t)(6000.0f / scale);
    if (scroll_period < 1500) scroll_period = 1500;
    int scroll = (int)((now % scroll_period) * 30 / scroll_period);

    for (int gx = -scroll; gx < w; gx += 30) {
        if (gx + x >= x && gx + x < x + w)
            d.drawFastVLine(x + gx, y, h, T_DIM);
    }
    for (int gy = -scroll; gy < h; gy += 30) {
        if (gy + y >= y && gy + y < y + h)
            d.drawFastHLine(x, y + gy, w, T_DIM);
    }

    /* Packet: hops along a precomputed L-path through the grid.
     * 4-second cycle. Magenta pulse (T_ACCENT2). */
    uint32_t pkt_period = (uint32_t)(4000.0f / scale);
    if (pkt_period < 800) pkt_period = 800;
    uint32_t pphase = now % pkt_period;
    /* L-path: 0-25% goes right, 25-50% goes down, 50-75% right, 75-100% down */
    int px, py;
    int seg = pphase * 4 / pkt_period;
    int seg_phase = (pphase * 4) % pkt_period;
    int seg_pct = (int)((int64_t)seg_phase * 100 / pkt_period);
    switch (seg) {
        case 0:  px = (w * 10 + (w * 30 - w * 10) * seg_pct / 100) / 100; py = h * 20 / 100; break;
        case 1:  px = w * 30 / 100; py = (h * 20 + (h * 75 - h * 20) * seg_pct / 100) / 100; break;
        case 2:  px = (w * 30 + (w * 75 - w * 30) * seg_pct / 100) / 100; py = h * 75 / 100; break;
        default: px = w * 75 / 100; py = (h * 75 + (h * 95 - h * 75) * seg_pct / 100) / 100; break;
    }
    /* Draw 3x3 packet glyph at (px, py) */
    d.fillRect(x + px - 1, y + py - 1, 3, 3, T_ACCENT2);
    /* Halo */
    d.drawPixel(x + px - 2, y + py, T_ACCENT2);
    d.drawPixel(x + px + 2, y + py, T_ACCENT2);
    d.drawPixel(x + px, y + py - 2, T_ACCENT2);
    d.drawPixel(x + px, y + py + 2, T_ACCENT2);
}
```

```cpp
/* ---- PHANTOM: cyberpunk glyph flashes ---- */
static void amb_phantom(int x, int y, int w, int h)
{
    auto &d = M5Cardputer.Display;
    uint32_t now = millis();
    float scale = ambient_speed_scale();

    /* 4 glyph slots, each fades in/out on its own cycle.
     * Glyphs picked from a small set of cyberpunk-coded chars (NOT occult). */
    static const char GLYPHS[] = "@#$%^&*<>=?/\\";
    static const int GLYPH_COUNT = sizeof(GLYPHS) - 1;

    static const struct { uint8_t cx_pct, cy_pct; uint16_t period_ms; uint32_t seed; bool alt; } SLOTS[4] = {
        { 18, 25, 5000, 0xAAAA, false },
        { 60, 55, 6000, 0xBBBB, true  },
        { 35, 70, 5500, 0xCCCC, false },
        { 75, 20, 6500, 0xDDDD, true  },
    };
    for (int i = 0; i < 4; ++i) {
        uint32_t period = (uint32_t)(SLOTS[i].period_ms / scale);
        if (period < 1500) period = 1500;
        uint32_t phase = (now + SLOTS[i].seed) % period;
        /* Triangle alpha 0..1..0 across the period */
        int alpha = (int)((phase * 200 / period));
        if (alpha > 100) alpha = 200 - alpha;
        if (alpha < 30) continue;  /* too dim to bother drawing */

        /* Pick glyph deterministically from seed + cycle count */
        uint32_t cycle = (now + SLOTS[i].seed) / period;
        char glyph = GLYPHS[(SLOTS[i].seed ^ cycle) % GLYPH_COUNT];
        uint16_t color = SLOTS[i].alt ? T_ACCENT2 : T_ACCENT;
        /* Fade with text size ramp — bigger when at peak alpha */
        int gx = x + (SLOTS[i].cx_pct * w) / 100;
        int gy = y + (SLOTS[i].cy_pct * h) / 100;
        d.setTextSize(2);
        d.setTextColor(color, T_BG);
        d.setCursor(gx, gy);
        d.print(glyph);
        d.setTextSize(1);  /* restore */
    }
}
```

```cpp
/* Mood color modulation. Returns blended color toward the mood's tint.
 * Blend amount: 0=no tint, 255=full mood color. We use ~128 for visible
 * but not overpowering shift. */
static uint16_t blend_565(uint16_t a, uint16_t b, uint8_t t)
{
    uint8_t ar = (a >> 11) & 0x1F, ag = (a >> 5) & 0x3F, ab = a & 0x1F;
    uint8_t br = (b >> 11) & 0x1F, bg = (b >> 5) & 0x3F, bb = b & 0x1F;
    uint8_t r = (ar * (255 - t) + br * t) / 255;
    uint8_t g = (ag * (255 - t) + bg * t) / 255;
    uint8_t bl = (ab * (255 - t) + bb * t) / 255;
    return (r << 11) | (g << 5) | bl;
}

/* Mood tint targets, RGB565. */
#define MOOD_TINT_HUNT     0xF1AC  /* warm aggressive #ff2266 */
#define MOOD_TINT_STEALTH  0x3175  /* cool muted #3060a0 */
#define MOOD_TINT_SURGICAL 0xFFFF  /* sharp white */
#define MOOD_TINT_STORM    0xFD40  /* amber #ffaa00 */

#define MOOD_BLEND_AMOUNT  140    /* ~55% toward tint */

uint16_t theme_effective_accent(void)
{
    extern int triton_current_mode_int(void);
    uint16_t base = theme().accent;
    switch (triton_current_mode_int()) {
        case 1:  return blend_565(base, MOOD_TINT_HUNT,     MOOD_BLEND_AMOUNT);
        case 2:  return blend_565(base, MOOD_TINT_STEALTH,  MOOD_BLEND_AMOUNT);
        case 3:  return blend_565(base, MOOD_TINT_SURGICAL, 100);  /* lighter blend */
        case 4:  return blend_565(base, MOOD_TINT_STORM,    MOOD_BLEND_AMOUNT);
        default: return base;
    }
}

{
    extern int triton_current_mode_int(void);

    switch (triton_current_mode_int()) {
        case 1:  return blend_565(base, MOOD_TINT_HUNT,     MOOD_BLEND_AMOUNT);
```

```python
"""
Slice a sprite sheet into N frames, downscale each to TARGET_W x TARGET_H,
quantize to 16 colors per theme, emit a C header with PROGMEM 4-bit
indexed arrays + per-theme palette LUT.

Usage:
    python tools/sprite_sheet_to_4bit.py <input.png> <output.h> \
        --cols 4 --rows 2 --target 48x40 \
        --names IDLE,HUNT_SCAN_0,HUNT_SCAN_1,ALERT,ATTACK,SLEEPY,STOKED,DESPAIR

Each frame is packed as 2 pixels per byte (high nibble = left pixel).
Output palettes: one set per theme defined below (POSEIDON, PHANTOM, MATRIX,
AMBER, TRON, EINK, HICONTRAST), each 16 RGB565 entries.
"""
import sys
import argparse
from PIL import Image

# Per-theme palette anchors. Each palette is 16 colors picked to map
# the sprite's tonal range to the theme's accent palette.
THEME_PALETTES = {
    "POSEIDON":   [(0,0,0),(8,8,16),(16,16,32),(32,32,48),(48,48,80),
                   (0,80,120),(0,120,180),(0,180,229),(120,180,229),
                   (200,80,140),(229,96,208),(255,96,208),(255,140,200),
                   (180,120,80),(220,180,120),(255,255,255)],
    "PHANTOM":    [(0,0,0),(10,4,18),(20,8,32),(40,16,60),(60,24,90),
                   (80,40,140),(120,60,180),(160,80,220),(176,112,255),
                   (200,80,180),(255,96,208),(255,140,220),(180,120,160),
                   (220,180,200),(140,140,140),(255,255,255)],
    "MATRIX":     [(0,0,0),(0,8,0),(0,16,0),(0,32,0),(0,48,0),
                   (0,80,0),(0,120,0),(0,180,0),(0,229,0),(80,229,80),
                   (120,255,120),(180,255,180),(60,90,60),(100,140,100),
                   (140,180,140),(255,255,255)],
    "AMBER":      [(0,0,0),(16,8,0),(32,16,0),(48,24,0),(64,32,0),
                   (96,48,0),(140,72,0),(180,100,0),(220,140,0),
                   (255,160,40),(255,200,80),(255,240,140),(140,80,20),
                   (180,120,60),(80,40,0),(255,255,255)],
    "TRON":       [(0,0,0),(0,4,16),(0,8,32),(0,16,48),(0,32,80),
                   (0,80,120),(0,120,180),(0,234,255),(120,200,255),
                   (255,0,140),(255,80,180),(255,140,200),(180,80,140),
                   (140,140,180),(60,80,120),(255,255,255)],
    "EINK":       [(255,255,255),(240,240,235),(220,220,210),(200,200,190),
                   (180,180,170),(160,160,150),(140,140,130),(120,120,110),
                   (100,100,90),(80,80,70),(60,60,55),(48,48,42),
                   (32,32,28),(20,20,18),(8,8,6),(0,0,0)],
    "HICONTRAST": [(0,0,0),(255,255,255),(255,0,0),(0,255,0),(0,0,255),
                   (255,255,0),(0,255,255),(255,0,255),(128,128,128),
                   (255,128,0),(128,255,0),(0,128,255),(255,0,128),
                   (128,0,255),(0,255,128),(192,192,192)],
}
THEME_ORDER = ["POSEIDON","PHANTOM","MATRIX","AMBER","TRON","EINK","HICONTRAST"]

def rgb_to_565(r, g, b):
    return ((r & 0xF8) << 8) | ((g & 0xFC) << 3) | (b >> 3)

def closest_index(rgb, palette):
    """Find closest palette entry by squared distance."""
    r, g, b = rgb
    best_i, best_d = 0, 999_999_999
    for i, (pr, pg, pb) in enumerate(palette):
        d = (r - pr) ** 2 + (g - pg) ** 2 + (b - pb) ** 2
        if d < best_d:
            best_d, best_i = d, i
    return best_i

def quantize_frame(frame, palette):
    """Return list of indices (0..15) row-major, packed 2 per byte."""
    w, h = frame.size
    px = frame.load()
    indices = []
    for y in range(h):
        for x in range(w):
            indices.append(closest_index(px[x, y][:3], palette))
    # pack 2px per byte (high nibble = even-x pixel)
    packed = bytearray()
    for i in range(0, len(indices), 2):
        hi = indices[i] & 0xF
        lo = indices[i+1] & 0xF if i+1 < len(indices) else 0
        packed.append((hi << 4) | lo)
    return packed

def main():
    ap = argparse.ArgumentParser()
    ap.add_argument("input", help="Sprite sheet PNG")
    ap.add_argument("output", help="Output .h file")
    ap.add_argument("--cols", type=int, required=True)
    ap.add_argument("--rows", type=int, required=True)
    ap.add_argument("--target", required=True, help="WxH e.g. 48x40")
    ap.add_argument("--names", required=True, help="Comma-separated frame names")
    ap.add_argument("--prefix", default="TRI_", help="Frame enum prefix")
    args = ap.parse_args()

    tw, th = (int(s) for s in args.target.split("x"))
    names = args.names.split(",")
    expected = args.cols * args.rows
    assert len(names) == expected, f"--names must have {expected} entries, got {len(names)}"

    sheet = Image.open(args.input).convert("RGB")
    sw, sh = sheet.size
    cell_w = sw // args.cols
    cell_h = sh // args.rows

    frames = []
    for r in range(args.rows):
        for c in range(args.cols):
            cell = sheet.crop((c*cell_w, r*cell_h, (c+1)*cell_w, (r+1)*cell_h))
            cell = cell.resize((tw, th), Image.LANCZOS)
            frames.append(cell)

    with open(args.output, "w", encoding="utf-8") as f:
        f.write(f"/* Auto-generated from {args.input}. Do not hand-edit. */\n")
        f.write(f"/* {len(frames)} frames @ {tw}x{th}, 4-bit indexed, packed 2px/byte. */\n\n")
        f.write("#pragma once\n#include <Arduino.h>\n#include <pgmspace.h>\n\n")

        # Frame enum
        f.write(f"enum {args.prefix.lower()}frame_t {{\n")
        for n in names:
            f.write(f"    {args.prefix}{n},\n")
        f.write(f"    {args.prefix}_COUNT\n}};\n\n")

        # Geometry
        f.write(f"#define {args.prefix}W {tw}\n")
        f.write(f"#define {args.prefix}H {th}\n")
        f.write(f"#define {args.prefix}BYTES_PER_FRAME (({args.prefix}W * {args.prefix}H) / 2)\n\n")

        # Per-theme palettes (RGB565)
        f.write(f"extern const uint16_t {args.prefix.lower()}palette[][16] PROGMEM;\n")
        f.write("#ifdef TRITON_SPRITES_IMPL\n")
        f.write(f"const uint16_t {args.prefix.lower()}palette[][16] PROGMEM = {{\n")
        for theme in THEME_ORDER:
            pal = THEME_PALETTES[theme]
            entries = ", ".join(f"0x{rgb_to_565(*c):04X}" for c in pal)
            f.write(f"    {{ {entries} }},  /* {theme} */\n")
        f.write("};\n")
        f.write("#endif\n\n")

        # Per-theme quantized frame data
        f.write(f"extern const uint8_t {args.prefix.lower()}sprite_data[][{args.prefix}_COUNT][{args.prefix}BYTES_PER_FRAME] PROGMEM;\n")
        f.write("#ifdef TRITON_SPRITES_IMPL\n")
        f.write(f"const uint8_t {args.prefix.lower()}sprite_data[][{args.prefix}_COUNT][{args.prefix}BYTES_PER_FRAME] PROGMEM = {{\n")
        for theme in THEME_ORDER:
            pal = THEME_PALETTES[theme]
            f.write(f"    {{  /* {theme} */\n")
            for frame_idx, frame in enumerate(frames):
                packed = quantize_frame(frame, pal)
                bytes_str = ", ".join(f"0x{b:02X}" for b in packed)
                f.write(f"        {{ {bytes_str} }},  /* {names[frame_idx]} */\n")
            f.write("    },\n")
        f.write("};\n")
        f.write("#endif\n")

if __name__ == "__main__":
    main()
```

```cpp
 * resolved from current theme. */
static void blit_triton_sprite(int cx, int cy, int frame_idx)
{
    auto &d = M5Cardputer.Display;
    if (frame_idx < 0 || frame_idx >= TRI__COUNT) return;
    int theme_idx = (int)theme_current_id();
    if (theme_idx < 0 || theme_idx >= 7) theme_idx = 0;
    int x0 = cx - TRI_W / 2;
    int y0 = cy - TRI_H / 2;
    for (int y = 0; y < TRI_H; ++y) {
        for (int x = 0; x < TRI_W; x += 2) {
            int byte_idx = (y * TRI_W + x) / 2;
            uint8_t b = pgm_read_byte(&tri_sprite_data[theme_idx][frame_idx][byte_idx]);
            uint8_t hi_idx = (b >> 4) & 0xF;
            uint8_t lo_idx = b & 0xF;
            uint16_t hi_color = pgm_read_word(&tri_palette[theme_idx][hi_idx]);
```

```python
"""
Convert an animated GIF (or PNG sequence) to a C header of 4-bit indexed
frame arrays + shared 16-color palette. Designed for Cardputer screensaver
playback at ~10-15 FPS.

Usage:
    python tools/gif_to_frames_4bit.py <input.gif> <output.h> <symbol_name> \
        [--target 120x67] [--fps 12] [--max-frames 24]
"""
import sys
import argparse
from PIL import Image, ImageSequence

def rgb_to_565(r, g, b):
    return ((r & 0xF8) << 8) | ((g & 0xFC) << 3) | (b >> 3)

def main():
    ap = argparse.ArgumentParser()
    ap.add_argument("input")
    ap.add_argument("output")
    ap.add_argument("symbol")
    ap.add_argument("--target", default="120x67")
    ap.add_argument("--fps", type=int, default=12)
    ap.add_argument("--max-frames", type=int, default=24)
    args = ap.parse_args()

    tw, th = (int(s) for s in args.target.split("x"))
    src = Image.open(args.input)

    # Frame sample: subsample if source has more than max-frames
    frames_raw = []
    for fr in ImageSequence.Iterator(src):
        frames_raw.append(fr.convert("RGB"))
    if len(frames_raw) > args.max_frames:
        step = len(frames_raw) / args.max_frames
        frames_raw = [frames_raw[int(i * step)] for i in range(args.max_frames)]

    # Resize all frames
    frames = [f.resize((tw, th), Image.LANCZOS) for f in frames_raw]

    # Build a global 16-color palette by quantizing the first frame
    palette_img = frames[0].quantize(colors=16, method=Image.MAXCOVERAGE).convert("P")
    raw_pal = palette_img.getpalette()[:48]   # 16 RGB triplets
    palette = [(raw_pal[i], raw_pal[i+1], raw_pal[i+2]) for i in range(0, 48, 3)]

    # Quantize each frame against the global palette and pack 2 px/byte
    def quant(frame):
        # Use ImagePalette dither
        flat = []
        px = frame.load()
        for y in range(th):
            for x in range(tw):
                r, g, b = px[x, y][:3]
                # Closest palette entry
                best, best_d = 0, 999_999_999
                for i, (pr, pg, pb) in enumerate(palette):
                    d = (r-pr)**2 + (g-pg)**2 + (b-pb)**2
                    if d < best_d: best_d, best = d, i
                flat.append(best)
        # Pack 2 px/byte
        packed = bytearray()
        for i in range(0, len(flat), 2):
            hi = flat[i] & 0xF
            lo = flat[i+1] & 0xF if i+1 < len(flat) else 0
            packed.append((hi << 4) | lo)
        return packed

    quantized = [quant(f) for f in frames]

    with open(args.output, "w", encoding="utf-8") as f:
        f.write(f"/* Auto-generated from {args.input}. Do not hand-edit. */\n")
        f.write(f"/* {len(frames)} frames @ {tw}x{th}, 4-bit indexed, {args.fps} FPS playback. */\n\n")
        f.write("#pragma once\n#include <Arduino.h>\n#include <pgmspace.h>\n\n")

        f.write(f"#define {args.symbol.upper()}_W {tw}\n")
        f.write(f"#define {args.symbol.upper()}_H {th}\n")
        f.write(f"#define {args.symbol.upper()}_FRAMES {len(frames)}\n")
        f.write(f"#define {args.symbol.upper()}_FRAME_DELAY_MS {1000 // args.fps}\n")
        f.write(f"#define {args.symbol.upper()}_BYTES_PER_FRAME (({tw} * {th}) / 2)\n\n")

        # Palette (RGB565)
        pal_565 = ", ".join(f"0x{rgb_to_565(*c):04X}" for c in palette)
        f.write(f"const uint16_t PROGMEM {args.symbol}_palette[16] = {{ {pal_565} }};\n\n")

        # Frame data
        f.write(f"const uint8_t PROGMEM {args.symbol}_frames[{len(frames)}][{tw*th//2}] = {{\n")
        for i, packed in enumerate(quantized):
            bytes_str = ", ".join(f"0x{b:02X}" for b in packed)
            f.write(f"    {{ {bytes_str} }},  /* frame {i} */\n")
        f.write("};\n")

if __name__ == "__main__":
    main()
```

```cpp
#include <M5Cardputer.h>
#include "../input.h"
#include "../assets/screensaver_hokusai.h"

void sav_hokusai_run(void)
{
    auto &d = M5Cardputer.Display;
    /* Centered, 2x scaled to fill more of 240x135 */
    const int dst_w = HOKUSAI_W * 2;
    const int dst_h = HOKUSAI_H * 2;
    const int x0 = (240 - dst_w) / 2;
    const int y0 = (135 - dst_h) / 2;
    int frame = 0;
    uint32_t last = 0;
    d.fillScreen(0x0000);

    while (true) {
        if (input_poll() != PK_NONE) break;
        uint32_t now = millis();
        if (now - last >= HOKUSAI_FRAME_DELAY_MS) {
            last = now;
            /* Blit frame: each byte is 2 pixels */
            for (int y = 0; y < HOKUSAI_H; ++y) {
                for (int x = 0; x < HOKUSAI_W; x += 2) {
                    int byte_idx = (y * HOKUSAI_W + x) / 2;
                    uint8_t b = pgm_read_byte(&hokusai_frames[frame][byte_idx]);
                    uint16_t hi = pgm_read_word(&hokusai_palette[(b >> 4) & 0xF]);
                    uint16_t lo = pgm_read_word(&hokusai_palette[b & 0xF]);
                    /* 2x scale */
                    d.fillRect(x0 + x*2,     y0 + y*2, 2, 2, hi);
                    d.fillRect(x0 + (x+1)*2, y0 + y*2, 2, 2, lo);
                }
            }
            frame = (frame + 1) % HOKUSAI_FRAMES;
        }
        delay(5);
    }
}
```
